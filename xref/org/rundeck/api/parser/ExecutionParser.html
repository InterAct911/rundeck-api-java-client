16  package org.rundeck.api.parser;
17  
18  import java.util.Collections;
19  import java.util.Date;
20  import java.util.HashSet;
21  import java.util.List;
22  
23  import org.apache.commons.lang.StringUtils;
24  import org.dom4j.Node;
25  import org.rundeck.api.domain.RundeckExecution;
26  import org.rundeck.api.domain.RundeckJob;
27  import org.rundeck.api.domain.RundeckExecution.ExecutionStatus;
28  import org.rundeck.api.domain.RundeckNode;
29  import org.rundeck.api.domain.RundeckNodeIdentity;
30  
31  /**
32   * Parser for a single {@link RundeckExecution}
33   * 
34   * @author Vincent Behar
35   */
36  public class ExecutionParser implements XmlNodeParser<RundeckExecution> {
37  
38      private String xpath;
39  
40      public ExecutionParser() {
41          super();
42      }
43  
44      /**
45       * @param xpath of the execution element if it is not the root node
46       */
47      public ExecutionParser(String xpath) {
48          super();
49          this.xpath = xpath;
50      }
51  
52      @Override
53      public RundeckExecution parseXmlNode(Node node) {
54          Node execNode = xpath != null ? node.selectSingleNode(xpath) : node;
55  
56          RundeckExecution execution = new RundeckExecution();
57  
58          execution.setId(Long.valueOf(execNode.valueOf("@id")));
59          execution.setUrl(StringUtils.trimToNull(execNode.valueOf("@href")));
60          try {
61              execution.setStatus(ExecutionStatus.valueOf(StringUtils.replace(StringUtils.upperCase(execNode.valueOf("@status")),"-","_")));
62          } catch (IllegalArgumentException e) {
63              execution.setStatus(null);
64          }
65          execution.setDescription(StringUtils.trimToNull(execNode.valueOf("description")));
66          execution.setArgstring(StringUtils.trimToNull(execNode.valueOf("argstring")));
67          execution.setStartedBy(StringUtils.trimToNull(execNode.valueOf("user")));
68          execution.setAbortedBy(StringUtils.trimToNull(execNode.valueOf("abortedby")));
69          execution.setProject(StringUtils.trimToNull(execNode.valueOf("@project")));
70          String startedAt = StringUtils.trimToNull(execNode.valueOf("date-started/@unixtime"));
71          if (startedAt != null) {
72              execution.setStartedAt(new Date(Long.valueOf(startedAt)));
73          }
74          String endedAt = StringUtils.trimToNull(execNode.valueOf("date-ended/@unixtime"));
75          if (endedAt != null) {
76              execution.setEndedAt(new Date(Long.valueOf(endedAt)));
77          }
78  
79          Node jobNode = execNode.selectSingleNode("job");
80          if (jobNode != null) {
81              RundeckJob job = new JobParser().parseXmlNode(jobNode);
82              execution.setJob(job);
83          }
84  
85          final Node successfulNodes = execNode.selectSingleNode("successfulNodes");
86          if (successfulNodes != null) {
87              final List<RundeckNode> rundeckNodes =
88                      new ListParser<RundeckNode>(new NodeParser(), "successfulNodes/node")
89                              .parseXmlNode(execNode);
90              execution.setSuccessfulNodes(new HashSet<RundeckNodeIdentity>(rundeckNodes));
91          }else{
92              execution.setSuccessfulNodes(Collections.<RundeckNodeIdentity>emptySet());
93          }
94  
95          final Node failedNodes = execNode.selectSingleNode("failedNodes");
96          if (failedNodes != null) {
97              final List<RundeckNode> rundeckNodes =
98                      new ListParser<RundeckNode>(new NodeParser(), "failedNodes/node")
99                              .parseXmlNode(execNode);
100             execution.setFailedNodes(new HashSet<RundeckNodeIdentity>(rundeckNodes));
101         } else {
102             execution.setFailedNodes(Collections.<RundeckNodeIdentity>emptySet());
103         }
104 
105         return execution;
106     }

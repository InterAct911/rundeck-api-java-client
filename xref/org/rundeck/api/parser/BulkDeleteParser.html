35  /**
36   * BulkDeleteParser is ...
37   *
38   * @author Greg Schueler <a href="mailto:greg@dtosolutions.com">greg@dtosolutions.com</a>
39   */
40  public class BulkDeleteParser implements XmlNodeParser<RundeckJobDeleteBulk> {
41      private String xpath;
42  
43      public BulkDeleteParser(String xpath) {
44          this.xpath = xpath;
45      }
46  
47      public BulkDeleteParser() {
48      }
49  
50      @Override
51      public RundeckJobDeleteBulk parseXmlNode(Node node) {
52          Node subnode = xpath != null ? node.selectSingleNode(xpath) : node;
53          final ArrayList<RundeckJobDelete> deletes = new ArrayList<RundeckJobDelete>();
54          final List results = subnode.selectNodes("(succeeded|failed)/deleteJobResult");
55          final DeleteParser parser = new DeleteParser();
56          if (null != results && results.size() > 0) {
57              for (final Object o : results) {
58                  deletes.add(parser.parseXmlNode((Node) o));
59              }
60          }
61  
62          final String requestcount = StringUtils.trimToNull(subnode.valueOf("@requestCount"));
63          final String allsuccessString = StringUtils.trimToNull(subnode.valueOf("@allsuccessful"));
64          int count = 0;
65          boolean allsuccess = false;
66          if (null != requestcount) {
67              try {
68                  count = Integer.parseInt(requestcount);
69              } catch (NumberFormatException e) {
70  
71              }
72          }
73          if (null != allsuccessString) {
74              allsuccess = Boolean.parseBoolean(allsuccessString);
75          }
76  
77          return new RundeckJobDeleteBulk(deletes, count, allsuccess);
78      }
79  }
